Table EMPRESA {
  ID_EMPRESA INTEGER [pk]
  RAZAO_SOCIAL VARCHAR
  NOME_FANTASIA VARCHAR
  CNPJ VARCHAR
  DATA_CRIACAO DATE
  SITUACAO VARCHAR
}

Table COLABORADOR {
  ID_COLABORADOR INTEGER [pk]
  NOME VARCHAR
  FUNCAO VARCHAR
  DATA_CRIACAO DATE
  SITUACAO VARCHAR
  ID_EMPRESA INTEGER [ref: > EMPRESA.ID_EMPRESA]
}

Table RESPONSAVEL_TECNICO_VENDA {
  ID_RESPONSAVEL_TECNICO_VENDA INTEGER [pk]
  RAZAO_SOCIAL VARCHAR
  NOME_FANTASIA VARCHAR
  CNPJ VARCHAR
  SITUACAO VARCHAR
  ID_EMPRESA INTEGER [ref: > EMPRESA.ID_EMPRESA]
  ID_ENDERECO INTEGER [ref: - ENDERECO.ID_ENDERECO]
}

Table ENDERECO {
  ID_ENDERECO INTEGER [pk]
  LOGRADOURO VARCHAR
  NUMERO VARCHAR
  COMPLEMENTO VARCHAR
  BAIRRO VARCHAR
  CEP VARCHAR
  CIDADE VARCHAR
  UF VARCHAR
}

Table CLIENTE {
  ID_CLIENTE INTEGER [pk]
  NOME VARCHAR
  ATIVIDADE_PRINCIPAL VARCHAR
  EMAIL VARCHAR
  TELEFONE VARCHAR
  DATA_CRIACAO DATE
  SITUACAO VARCHAR
  ID_TIPO_CLIENTE INTEGER [ref: > TIPO_CLIENTE.ID_TIPO_CLIENTE]
  ID_ENDERECO INTEGER [ref: - ENDERECO.ID_ENDERECO]
  ID_DOCUMENTO INTEGER [ref: - DOCUMENTO.ID_DOCUMENTO]
}

Table DOCUMENTO {
  ID_DOCUMENTO INTEGER [pk]
  NUMERO VARCHAR
}

Table TIPO_CLIENTE {
  ID_TIPO_CLIENTE INTEGER [pk]
  DESCRICAO VARCHAR
  ABREVIACAO VARCHAR
}

Table FAZENDA {
  ID_FAZENDA INTEGER [pk]
  DESCRICAO VARCHAR
  TAMANHO_HA INTEGER
  DATA_SINTEGRA DATE
  INSCRICAO_ESTADUAL VARCHAR
  DATA_CRIACAO DATE
  SITUACAO VARCHAR
  ID_CLIENTE INTEGER [ref: > CLIENTE.ID_CLIENTE]
  ID_ENDERECO INTEGER [ref: - ENDERECO.ID_ENDERECO]
}

Table PRODUTO {
  ID_PRODUTO INTEGER [pk]
  DESCRICAO VARCHAR
  UNIDADE VARCHAR
  PRODUTO_CONTROLADO_POR_LOTE BOOLEAN
  PESO DOUBLE
  SITUACAO VARCHAR
  ID_TIPO_PRODUTO INTEGER [ref: > TIPO_PRODUTO.ID_TIPO_PRODUTO]
}

Table TIPO_PRODUTO {
  ID_TIPO_PRODUTO INTEGER [pk]
  DESCRICAO VARCHAR
  ABREVIACAO VARCHAR
}

Table PEDIDO_VENDA {
  ID_PEDIDO_VENDA INTEGER [pk]
  NUMERO INTEGER
  VALOR DECIMAL
  STATUS INTEGER
  TIPO_VENDA VARCHAR
  FORMA_PAGAMENTO VARCHAR
  TIPO_FRETE INTEGER
  ID_CLIENTE INTEGER [ref: > CLIENTE.ID_CLIENTE]
  ID_FAZENDA INTEGER [ref: > FAZENDA.ID_FAZENDA]
  ID_EMPRESA INTEGER [ref: > EMPRESA.ID_EMPRESA]
  ID_RESPONSAVEL_TECNICO_VENDA INTEGER [ref: > RESPONSAVEL_TECNICO_VENDA.ID_RESPONSAVEL_TECNICO_VENDA]
}

Table ITEM_PEDIDO_VENDA {
  ID_ITEM_PEDIDO_VENDA INTEGER [pk]
  QUANTIDADE DOUBLE
  VALOR_UNITARIO DECIMAL
  VALOR_TOTAL DECIMAL
  ID_PEDIDO_VENDA INTEGER [ref: > PEDIDO_VENDA.ID_PEDIDO_VENDA]
  ID_PRODUTO INTEGER [ref: > PRODUTO.ID_PRODUTO]
}
